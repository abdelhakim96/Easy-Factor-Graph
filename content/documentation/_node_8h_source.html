125  T* Insert_with_size_check(T* pot) { //returns the potential actually inserted in the graph (shape or exponential) cloned or not
126 
127  T* inserted_to_return = NULL;
128 
129  size_t var_numb = pot->Get_involved_var_safe()->size();
130  if (var_numb == 1) {
131  //unary potential insertion
132  Categoric_var* varU = pot->Get_involved_var_safe()->front();
133  auto itN = this->Nodes.begin();
134  bool node_found = false;
135 
136  for (itN; itN != this->Nodes.end(); itN++) {
137  if ((*itN)->Get_var()->Get_name().compare(varU->Get_name()) == 0) {
138 
139  if (this->bDestroy_Potentials_and_Variables)
140  inserted_to_return = new T(pot, { (*itN)->Get_var() });
141  else
142  inserted_to_return = pot;
143 
144  (*itN)->Permanent_Unary.push_back(new Potential(inserted_to_return));
145  node_found = true;
146  break;
147  }
148  }
149 
150  if (!node_found) {
151  system("ECHO the unary potential provided refers to an inexistent node");
152  abort();
153  }
154  }
155  else if (var_numb == 2) {
156  //binary potential insertion
157  Node* peer_A = NULL;
158  Node* peer_B = NULL;
159 
160  Categoric_var* varA = pot->Get_involved_var_safe()->front();
161  auto itN = this->Nodes.begin();
162  for (itN; itN != this->Nodes.end(); itN++) {
163  if ((*itN)->Get_var()->Get_name().compare(varA->Get_name()) == 0) {
164  peer_A = *itN;
165  break;
166  }
167  }
168  Categoric_var* varB = pot->Get_involved_var_safe()->back();
169  itN = this->Nodes.begin();
170  for (itN; itN != this->Nodes.end(); itN++) {
171  if ((*itN)->Get_var()->Get_name().compare(varB->Get_name()) == 0) {
172  peer_B = *itN;
173  break;
174  }
175  }
176 
177  if ((peer_A != NULL) && (peer_B != NULL)) {
178  //check this potential was not already inserted
179  const std::list<Categoric_var*>* temp_var;
180  for (auto it_bb = this->Binary_potentials.begin();
181  it_bb != this->Binary_potentials.end(); it_bb++) {
182  temp_var = (*it_bb)->Get_involved_var_safe();
183 
184  if (((peer_A->Get_var() == temp_var->front()) && (peer_B->Get_var() == temp_var->back())) ||
185  ((peer_A->Get_var() == temp_var->back()) && (peer_B->Get_var() == temp_var->front()))) {
186  system("ECHO found clone of an already inserted binary potential");
187  abort();
188  }
189  }
190  }
191 
192  if (peer_A == NULL) {
193  if (this->bDestroy_Potentials_and_Variables)
194  this->Nodes.push_back(new Node(varA));
195  else
196  this->Nodes.push_back(new Node(varA, true));
197  peer_A = this->Nodes.back();
198  }
199  if (peer_B == NULL) {
200  if (this->bDestroy_Potentials_and_Variables)
201  this->Nodes.push_back(new Node(varB));
202  else
203  this->Nodes.push_back(new Node(varB, true));
204  peer_B = this->Nodes.back();
205  }
206 
207  if (this->bDestroy_Potentials_and_Variables)
208  inserted_to_return = new T(pot, { peer_A->Get_var(), peer_B->Get_var() });
209  else
210  inserted_to_return = pot;
211 
212  Potential* new_pot = new Potential(inserted_to_return);
213  //create connection
214  Node::Neighbour_connection* A_B = new Node::Neighbour_connection();
215  A_B->This_Node = peer_A;
216  A_B->Neighbour = peer_B;
217  A_B->Shared_potential = new_pot;
218  A_B->Message_to_this_node = NULL;
219 
220  Node::Neighbour_connection* B_A = new Node::Neighbour_connection();
221  B_A->This_Node = peer_B;
222  B_A->Neighbour = peer_A;
223  B_A->Shared_potential = new_pot;
224  B_A->Message_to_this_node = NULL;
225 
226  A_B->Message_to_neighbour_node = &B_A->Message_to_this_node;
227  B_A->Message_to_neighbour_node = &A_B->Message_to_this_node;
228 
229  peer_A->Active_connections.push_back(A_B);
230  peer_B->Active_connections.push_back(B_A);
231 
232  this->Binary_potentials.push_back(new_pot);
233 
234  }
235  else {
236  system("ECHO invalid component to insert in a graph");
237  abort();
238  }
239 
240  if (this->Last_propag_info != NULL) delete this->Last_propag_info;
241  this->Last_propag_info = NULL;
242  this->mState = 0;
243 
244  return inserted_to_return;
