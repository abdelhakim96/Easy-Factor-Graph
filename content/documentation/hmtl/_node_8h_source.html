237  template<typename T>
238  T* ___Insert(T* pot) { //returns a reference to the real potential inserted
239 
240  if (!pot->get_validity()) {
241 #ifdef _DEBUG
242  system("ECHO Insertion failed");
243 #endif // DEBUG
244  return NULL;
245  }
246 
247  T* inserted = NULL;
248  auto var_involved = pot->Get_involved_var_safe();
249  if (var_involved->size() == 1) {
250 //unary potential insertion
251  auto node = this->__Find_Node(var_involved->front());
252  if (node == NULL) {
253  if (this->Create_new_node(var_involved->front()))
254  this->Nodes.back()->Permanent_Unary.push_back(this->Get_for_insertion(pot, &inserted, { this->Nodes.back()->Get_var() }));
255  else return NULL;
256  }
257  else node->Permanent_Unary.push_back(this->Get_for_insertion(pot, &inserted, { node->Get_var() } ));
258  }
259  else if (var_involved->size() == 2) {
260 //binary potential insertion
261  Node* peer_A = this->__Find_Node(pot->Get_involved_var_safe()->front());
262  Node* peer_B = this->__Find_Node(pot->Get_involved_var_safe()->back());
263 
264  if ((peer_A != NULL) && (peer_B != NULL)) {
265  // check whether this binary potential was already present
266  auto var_A = peer_A->Get_var();
267  auto var_B = peer_B->Get_var();
268  const std::list<Categoric_var*>* vars_temp;
269  for (auto it = this->Binary_potentials.begin(); it != this->Binary_potentials.end(); it++) {
270  vars_temp = (*it)->Get_involved_var_safe();
271  if (( (vars_temp->front()->Get_name().compare(peer_A->Get_var()->Get_name()) == 0) && (vars_temp->back()->Get_name().compare(peer_B->Get_var()->Get_name()) == 0) ) ||
272  ((vars_temp->front()->Get_name().compare(peer_B->Get_var()->Get_name()) == 0) && (vars_temp->back()->Get_name().compare(peer_A->Get_var()->Get_name()) == 0))) {
273 #ifdef _DEBUG
274  system("ECHO found clone of an already inserted binary potential");
275 #endif // DEBUG
276  return NULL;
277  }
278  }
279  }
280 
281  if (peer_A == NULL) {
282  if (!this->Create_new_node(pot->Get_involved_var_safe()->front()))
283  return NULL;
284  peer_A = this->Nodes.back();
285  }
286  if (peer_B == NULL) {
287  if (!this->Create_new_node(pot->Get_involved_var_safe()->back()))
288  return NULL;
289  peer_B = this->Nodes.back();
290  }
291 
292  auto new_pot = this->Get_for_insertion(pot, &inserted, { peer_A->Get_var() , peer_B->Get_var() });
293 
294  //create connection
295  Node::Neighbour_connection* A_B = new Node::Neighbour_connection();
296  A_B->This_Node = peer_A;
297  A_B->Neighbour = peer_B;
298  A_B->Shared_potential = new_pot;
299  A_B->Message_to_this_node = NULL;
300 
301  Node::Neighbour_connection* B_A = new Node::Neighbour_connection();
302  B_A->This_Node = peer_B;
303  B_A->Neighbour = peer_A;
304  B_A->Shared_potential = new_pot;
305  B_A->Message_to_this_node = NULL;
306 
307  A_B->Message_to_neighbour_node = &B_A->Message_to_this_node;
308  B_A->Message_to_neighbour_node = &A_B->Message_to_this_node;
309 
310  peer_A->Active_connections.push_back(A_B);
311  peer_B->Active_connections.push_back(B_A);
312 
313  this->Binary_potentials.push_back(new_pot);
314 
315  }
316  else {
317 #ifdef _DEBUG
318  system("ECHO Only binary or unary potential can be inserted");
319 #endif // DEBUG
320  return NULL;
321  }
322 
323  if (this->Last_propag_info != NULL) delete this->Last_propag_info;
324  this->Last_propag_info = NULL;
325  this->mState = 0;
326  return inserted;
